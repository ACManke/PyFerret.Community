--------------------pattern_demo.jnl-------------
cancel data/all
cancel mode verify 
use coads_climatology
set region/l=1
shade slp

!show sym/all

!message "Using the symbols above we create some variables for the ppl shakey command."

let ylo = ($ppl$yorg)
let ymid = ylo + ($ppl$ylen)*3/5 ! 1010 is 3/5 of the way between 980 and 1030

let xlo = ($ppl$xorg) + ($ppl$xlen) + .2

ppl fill
fill/over/set_up/key/lev=(980,1010,5)/pal=bluescale/pat=6patterns slp

ppl fill/over

pattern solid
go fland
