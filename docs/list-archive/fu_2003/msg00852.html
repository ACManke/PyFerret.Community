curvilinear coordinates.  Is there a straight forward of accomplishing

dimensions:
        time = UNLIMITED ; // (72 currently)
        i_dot = 121 ;
        j_dot = 165 ;
        i_cross = 120 ;
        j_cross = 164 ;
        sigma_level = 25 ;
        sigma_level_full = 26 ;
        categories = 23 ;
        season = 2 ;

        float u(time, sigma_level, i_dot, j_dot) ;
                u:long_name = "U COMPONENT OF HORIZONTAL WIND" ;
