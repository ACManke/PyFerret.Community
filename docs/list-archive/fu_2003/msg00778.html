Hi Wendy,

After gathering the information across the help list I do:

USE sst_coads.cdf
LET sst_climatology = sst[GT=month_reg@MOD]
LET sst_anomaly=sst-sst_climatology[gt=sst]
LET sst_std = sst_anomaly / sst[x=10w:5w@var,y=34n:38n@var,l=1:456@var]^.5

Hope this helps
