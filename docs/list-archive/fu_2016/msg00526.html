and therefore it worked with this command

let p=pre[d=6,x=44.25E:51.75E@ave,y=2.25N:10.5S@ave,l=1:3@ave]-pr[d=2,x=44.25E:51.75E@ave,y=2.25N:10.5S@ave,l=1:3@ave]

list q
let s=q/3

list m
