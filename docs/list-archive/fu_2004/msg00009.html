repeat/l=1:12 (
set viewport v`l`;
contour/nolab/x=35w/y=1.5:13.5/z=0:40000/level="(-120,120,5)"/set us;
ppl axset,1,1,0,1;
ppl ylab,@sr;
ppl origin 1,1.2;
ppl axlen 2.7,2.7;
ppl xaxis 1.5,13.5,2;
ppl xfor (i2''N'');
ppl tics,,0.2,,,;
ppl contour;
ppl %yaxis/nouser,-40000,0,10000,0,,,100,-1,0.10,(i4);
label 8,2500,0,0,.14 "@CR MONTH `l`";
label -.1,-5000,0,0,.10 "@CR (m)";
contour/over/nolab/line/lev="(21.5)dark(21.5)"/color=black/x=35w/y=1.5:13.5/z=0:40000
ts)
