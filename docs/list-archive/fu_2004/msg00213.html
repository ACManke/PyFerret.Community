define symbol dir=data
define symbol label=test2
define symbol gridfile=($label)-grid.data
define symbol datafile=($label).data

define axis/x=1:2:1 xaxis
define axis/y=1:2:1 yaxis
define grid/x=xaxis/y=yaxis grid1

file/var=xcoo,zcoo/grid=grid1 "./($dir)/($gridfile)"
let xg=xcoo[d=($gridfile)]
let zg=zcoo[d=($gridfile)]

file/var=dill,sal,temp/grid=grid1 "./($dir)/($datafile)"
shade dill,xg,zg
