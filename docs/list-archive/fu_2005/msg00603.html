 define axis/t=1789:1992/npoints=204 t2
  define grid/t=t2 g2
  file/var=aa/grid=g2 d:\test.asc

  def view/xlim=0.2,0.82/ylim=0.4,0.7 v11

  plot/nolabel/line=8/set aa[l=1:204]

  ppl tics .08,.12,.08,.12
  ppl yfor "(i3)"
  ppl yaxis -15,15,2.5
  ppl axlsze,0.2,0.2

  ppl plot
