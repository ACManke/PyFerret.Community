use ohc.nc
use dens.nc

sh gr heat[d=1]

define axis/z=`RHO[x=@min,y=@min,l=@min,z=@min,d=2]`:`RHO[x=@max,y=@max,l=@max,z=@max,d=2]`:.1 axden

let heatsig=zaxreplace(HEAT[d=1],NEUTRALRHO[d=2,gx=heat[d=1],gy=heat[d=1],gt=heat[d=1]],z[gz=axden])
