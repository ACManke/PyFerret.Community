> gaorz wrote:
> 
> Hi,all of you
> 
> If I have sea surface wind field (u,v), how to obtain the curl of wind
> stress by Ferret?
> 
> Surpose U and V are 3D data ( X=100:125:1,  Y=-5: 25:1,
> T="10-apr-1998":"09-apr-1999":1 ).
> 

let rho_air = 1.2 ! kg/m^3 
let cd = 1.2e-3 ! non-dimensional drag
let/title="wind speed"/units="m/s" wmag = (u^2+v^2)^.5
let/title="Zonal Wind Stress"/units="N/m^2" taux = rho_air*cd*wmag*u
let/title="Meridional Wind Stress"/units="N/m^2" tauy =
rho_air*cd*wmag*v
let/title="windstress curl"/units="N/m^3" curl_tau =
(tauy[x=@ddc]-taux[y=@ddc]) 

If you're interested in curl_tau/f...

let pi = atan(1)*4
let omega = 7.292e-5
let f = sin(y[g=u[d=1]]*pi/180)*omega

let/title="windstress curl/f"/units="N/m^3/s" curl_tau_ovf = curl_tau/f
