let tt = t[gt=tax]
list tt[t=9-feb-2003] - tt[t=3-oct-2001]

or if you want to use this to define a label position, for example:

let tdiff = tt[t=9-feb-2003] - tt[t=3-oct-2001]

now you can use the number `tdiff`
