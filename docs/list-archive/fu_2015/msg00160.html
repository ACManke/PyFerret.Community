Try this sample example

[akshay@localhost tmp]$ cat sample.jnl
 ! NOAA/PMEL TMAP
 ! FERRET v6.93 
 ! Linux 2.6.32-504.el6.x86_64 64-bit - 11/13/14
 ! 23-Apr-15 15:13    

set data "http://oceanwatch.pifsc.noaa.gov/thredds/dodsC/aqua/monthly"
let points = t[g=chla]
list points
list l[g=points,t="01-FEB-2015"]
let xxx = l[g=points,t="01-FEB-2015"]
list xxx
q
